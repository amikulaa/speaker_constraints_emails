/**
 * Update the meta data attached to invitation for the sending of survey
 * link.
 * Updates sent_at value to current timestamp, and due date sent for
 * future reference.
 *
 * Returns true on success, false on error
 */
function update_invitation_meta($meta_value, $item_id, $field_id, $error_code){
    global $wpdb;
    $error_log = false;
    $error_code_section = '';
    // try to update first
    $updated = $wpdb->update(
        "{$wpdb->prefix}HIDDEN",
        ['meta_value' => $meta_value],
        ['item_id' => $item_id, 'field_id' => $field_id]
    );
    // check if update worked
    if ($updated === false) {
        $error_log = true;
        $error_code_section = 'UF';
    } else if ($updated === 0) {
        // likely missing row, try to insert
        $inserted = $wpdb->insert(
            "{$wpdb->prefix}HIDDEN", 
            [
                'item_id' => $item_id,
                'field_id' => $field_id,
                'meta_value' => $meta_value
            ]
        );
        // check if insert worked
        if ($inserted === false) {
            $error_log = true;
            $error_code_section = 'IF';
        }
    }
    // if error, show error msg
    if ($error_log) {
        $curr_query = $wpdb->last_query;
        $curr_error = $wpdb->last_error;
        error_log("SQL ERROR: speaker survey $error_code-$error_code_section: [ $curr_error ] with query [ $curr_query ]");
    }
    return $error_log == true ? false : true;
}
